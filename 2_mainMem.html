 84 //Shifts through all the bits in a given number to determine max bits needed for decmial version.
 85 int numberofbits(int num){
 86 
 87     int count = 0;
 88     
 89     while (num != 0)
 90     {
 91         count += 1;
 92         num = (num >> 1);
 93     }
 94         count = count -1;
 95         return count;
 96 }
