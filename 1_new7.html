 22 int main()
 23 {    
 24     char cont = 'y';
 25 
 26     while (cont == 'y')
 27     {
 28 
 29     int mainbytes, cachebytes, blocksize, N;
 30     char replacement, input[20];
 31 
 32 
 33     printf("\nEnter the size of the main memory in bytes:");
 34     scanf("%d", &mainbytes);    //Main memory size is stored in mainbytes.
 35     //printf("%d\n", mainbytes);
 36 
 37     printf("Enter the size of the cache in bytes:");
 38     scanf("%d", &cachebytes);   //Cache size is stored in cachebytes.
 39     //printf("%d\n", cachebytes);
 40 
 41     printf("Enter the cache block/line size:");
 42     scanf("%d", &blocksize);    //Block size is store in blocksize.
 43     //printf("%d\n", blocksize);
 44     
 45     printf("\nEnter the degree of set-associativity(input n for an n-way set-associative mapping):");
 46     scanf("%d", &N);            //N for associativity
 47     //printf("%d\n", N);
 48 
 49     printf("Enter the replacement policy (L=LRU, F=FIFO):");
 50     scanf(" %c", &replacement);  //Replacement holds the replacement policy for the cache.
 51     
 52     printf("Enter the name of the input file containing the list of memory references generated\n");
 53     printf("by the CPU: ");
 54     scanf("%s", input);          //Input stores the file name for the memory references.
 55 
 56     printf("Replacement: %c  Input File: %s\n", replacement,input); //For error checking
 57 
 58     mainMem *m = newMem(mainbytes);
 59 
 60     printf("\nSimulator output: \n");
 61     printf("Total address lines required = %d \n", addressLines(m));
 62 
 63     cache *c = newCache(addressLines(m),cachebytes,blocksize,N,replacement);
 64     printf("Number of bits for offset= %d \n", offBits(c));
 65     printf("Number of bits for index= %d \n", indexBits(c));
 66     printf("Number of bits for tag= %d \n", tagBits(c));
 67     printf("Total cache size required= %d \n\n", totCacheSize(c));
 68     printf("main memory address\t mmblk#\t cmset#\t cmblk#\t hit/miss\n");
 69     printf("_________________________________________________________________\n");
 70 
 71 /**********************************Read in Operations*********************************/
 72     FILE *fp;
 73     fp = fopen(input,"r");
 74 
 75     int num_of_addresses=0,i,j, count=0;
 76     fscanf(fp,"%d", &num_of_addresses);
 77 
 78     int address;  char operation;
 79     DArray *memOps = newDArray(displayMem);
 80 
 81     int mmblk_list[num_of_addresses];
 82 
 83     for (i = 0; i < num_of_addresses; ++i)
 84     {
 85         fscanf(fp," %c %d", &operation, &address);
 86         memLoc *loc = newMemLoc(i, operation,address,CacheBlockSize(c), CacheSets(c), Associativity(c));
 87         mmblk_list[i] = getMMBlk(loc);
 88         for (j=i-1; j >= 0; --j)
 89         {    
 90             if (mmblk_list[i] == mmblk_list[j])
 91             {
 92                 count++;
 93                 // loc->hit_miss = "hit";
 94                 break;
 95             }
 96         }
 97         insertDArray(memOps,loc);
 98     }
 99     fclose(fp);
100 
101 /************************************Creating cache***************************************/
102 
103     DArray *DcacheSets[CacheSets(c)];
104     queue *QcacheSets[CacheSets(c)];
105 
106     for (i = 0; i < CacheSets(c); ++i)
107     {
108         DcacheSets[i] = newDArray(displayC);//creates an array for each cache set
109         QcacheSets[i] = newQueue(displayC);//creates a queue for each cache set
110         for (j = 0; j < Associativity(c); ++j)
111         {
112             insertDArray(DcacheSets[i], NULL);//initializes the array with NULL
113         }
114     }
115 
116     int hits = 0;
117     for (i = 0; i < num_of_addresses; ++i)//loops through each address
118     {
119         int found = 0;
120         memLoc *Op = getDArray(memOps, i);//get the current operation from the darray
121         
122         //initializes a new cache block
123         cacheBlock *cblock = newCBlock(getMMOp(Op),getMMBlk(Op),getOpNum(Op),1, tagBits(c));
124         //sets the tag for the cache block
125         setTag(cblock, getAddress(Op), tagBits(c), addressLines(m));
126         //If there is an item in the array check the array for a hit
127         if (isNotEmpty(DcacheSets[getCSet(Op)]))  
128         {
129             for (j = 0; j < filledSlots(DcacheSets[getCSet(Op)]); ++j)
130             {    
131                 if (getMMBlk(Op) == getData(getDArray(DcacheSets[getCSet(Op)],j)))
132                 { 
133                     //if the address is found in the correct set then sets it as hit
134                     updateAge(getDArray(DcacheSets[getCSet(Op)],j),Op->opNum);
135                     Op->hit_miss="hit";
136                     found = 1;
137                     hits++;
138                     break;
139                 }
140             }
141         }
142         if (found != 1)   //If there were not hits 
143         {    //Set the next empty slot if the memory block isn't full
144             if (!(isFull(DcacheSets[getCSet(Op)])))  
145             {
146                 if (replacement == 'F')
147                 {
148                     //enqueue the operation for use with FIFO
149                     enqueue(QcacheSets[getCSet(Op)], cblock);
150                 }
151                 //Set next empty slot in CacheSet for DArray
152                 setDArray(DcacheSets[getCSet(Op)], filledSlots(DcacheSets[getCSet(Op)]), cblock);
153             }
154             //If cache set is full replace something  
155             //This conditional is the only place where the replacement policy matters
156             else     
157             {
158                 if (replacement == 'F')
159                 {    //dequeues the first operation preformed
160                     cacheBlock *removed = dequeue(QcacheSets[getCSet(Op)]);
161                     for (j = 0; j < Associativity(c); ++j)
162                     {    //break out if there is a hit
163                         if (getData(removed) == getData(getDArray(DcacheSets[getCSet(Op)],j)))
164                         {
165                             break;
166                         }
167                     }
168                     //if not hit, replace that position in the darray 
169                     setDArray(DcacheSets[getCSet(Op)], j, cblock);
170                     //and enqueue the new operation to track it
171                     enqueue(QcacheSets[getCSet(Op)], cblock);
172                 }
173                 //LIU replacement policy
174                 else if(replacement == 'L')
175                 {    //gets the age of the current cache block(it's OpNum)
176                     int smallest = getAge(cblock), index = 0;
177                     //loops through set
178                     for (j = 0; j < Associativity(c); ++j)
179                     {    //gets the age of the current operation
180                         int age = getAge(getDArray(DcacheSets[getCSet(Op)],j));
181                         //if the cache blocks age is greater than the new operation
182                         //then reset the cache blocks age and update index for the setDArray
183                         if (smallest > age)
184                         {
185                             smallest = age;
186                             index = j;
187                         }
188                     }
189                     //alwas inserts something if any conditionals above were met
190                     //then a new memory  block was inserted
191                     setDArray(DcacheSets[getCSet(Op)], index, cblock);
192                 }
193             }
194         }
195     }
196 
197     //Fills in untouched slots with default values
198     for (i = 0; i < CacheSets(c); ++i)
199     {
200         for (j = 0; j < Associativity(c); ++j)
201         {
202             if (getDArray(DcacheSets[i],j)==NULL)
203             {
204                 //initializes empty slots in the set
205                 cacheBlock *emptySlot = newCBlock('R',-1, -1,0, tagBits(c));
206                 setDArray(DcacheSets[i], j, emptySlot);
207             }
208         }
209     }
210     //Sets cacheblock numbers
211     int blockNumber = 0;
212     for (i = 0; i < CacheSets(c); ++i)
213     {
214         for (j = 0; j < Associativity(c); ++j)
215         {
216             //sets the cache blocks with correct data
217             cacheBlock *block = getDArray(DcacheSets[i],j);
218             setCBNum(block, blockNumber);
219             blockNumber++;
220         }
221     }
222 
223 /*********************************Print tables*********************************/
224 
225     displayDArray(stdout,memOps);//displays the memory table
226 
227     //caclculate the highest possible and actual hit rates
228     float hphr = ((float)count/(float)num_of_addresses)*100;
229     float ahr = ((float)hits/(float)num_of_addresses)*100;
230 
231     printf("\nHighest possible hit rate %d/%d = %.02f%%\n", count, num_of_addresses, hphr);
232     printf("Actual hit rate = %d/%d = %.02f%%\n\n", hits, num_of_addresses, ahr);
233 
234     printCacheHeading();
235     printf("_________________________________________________________________\n");
236 
237     for (i = 0; i < CacheSets(c); ++i)
238     {
239         displayDArray(stdout,DcacheSets[i]);
240     }
241 
242     free(c);//free cache
243     free(m);//free mem
244     free(*DcacheSets);
245     free(*QcacheSets);
246 
247     printf("Continue? (y = yes, n = no): \n");
248     scanf("\n%c", &cont);
249     }
250 
251     return 0;
252 }
253 //checks for an item in the array (lets us know if something has been written yet or not)
254 int isNotEmpty(DArray *array)
255 {
256     int i;
257     for (i = 0; i < sizeDArray(array); ++i)
258     {
259         if (getDArray(array, i)!=NULL)
260         {
261             return 1;
262         }
263     }
264     return 0;
265 }
266 //checks if the set is full
267 int isFull(DArray *array)
268 {
269     int i;
270     for (i = 0; i < sizeDArray(array); ++i)
271     {
272         if (getDArray(array, i)==NULL)
273         {
274             return 0;
275         }
276     }
277     return 1;
278 }
279 //returns which slots are filled in the darray
280 int filledSlots(DArray *array)
281 {
282     int i;
283     for (i = 0; i < sizeDArray(array); ++i)
284     {
285         if (getDArray(array,i)==NULL)
286         {
287             return i;
288         }
289     }
290     return sizeDArray(array);
291 }
